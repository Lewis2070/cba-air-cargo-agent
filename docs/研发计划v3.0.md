## 一、项目背景与目标

### 1.1 现状问题
- 评审组对当前交付版本不满意，核心原因：
  - 无真正的核心算法（重心平衡、装箱优化完全缺失）
  - 前端为静态 HTML，React 代码未构建
  - AI 功能为关键词 mock，非真正 NLP/优化算法
  - 交互逻辑不符合货运实际业务流程

### 1.2 核心目标
**单机计费重最大化** — 航司商业价值核心

```
计费重 = max(实际毛重, 体积重)
体积重 = 长(cm) × 宽(cm) × 高(cm) ÷ 6000

密度 > 167 kg/m³ → 按毛重计费 = 优先装
密度 < 167 kg/m³ → 按体积计费 = 优先后排
```

### 1.3 核心优化目标函数

```
Max: Σ(计费重_i × x[i][j])        ← 最大化单机总计费重

约束:
  ① Σ毛重_i ≤ 机型最大业载(kg)     ← 重量限制
  ② Σ体积_i ≤ 机型总容积(m³)       ← 体积限制
  ③ CG_index ∈ [CG_min, CG_max]   ← 安全重心
  ④ DGR隔离规则                     ← 危险品规则
  ⑤ ULD类型匹配                     ← 板型匹配
  ⑥ 单ULD内重量/体积 ≤ ULD限值      ← 打板限制
```

### 6.2 重心计算公式

```typescript
// 重心位置(m) = Σ(重量_i × 力臂_i) / Σ重量_i
// 重心指数 = (重心位置 - MAC前缘) / MAC × 100

interface CGResult {
  cgPosition: number;      // 重心位置 (m)
  cgIndex: number;         // 重心指数 (%)
  totalWeight: number;      // 总重量 (kg)
  totalMoment: number;      // 总力矩 (kg·m)
  isInLimits: boolean;      // 是否在安全范围内
  isWarning: boolean;       // 是否接近极限
  limitMin: number;         // 下限
  limitMax: number;         // 上限
  advice: string;           // 调整建议
}

function calculateCG(placements: UldPlacement[], aircraft: AircraftConfig): CGResult {
  const totalMoment = placements.reduce(
    (sum, p) => sum + p.weight * p.position.arm, 0
  );
  const totalWeight = placements.reduce((sum, p) => sum + p.weight, 0);
  const cgPosition = totalWeight > 0 ? totalMoment / totalWeight : 0;

  const cgIndex = ((cgPosition - aircraft.macLeadingEdge) / aircraft.mac) * 100;
  const limits = aircraft.cgLimits.takeoff;

  const isInLimits = cgIndex >= limits[0] && cgIndex <= limits[1];
  const isWarning = cgIndex < limits[0] + 3 || cgIndex > limits[1] - 3;

  let advice = '';
  if (!isInLimits) {
    if (cgIndex < limits[0]) {
      advice = `重心偏前(${cgIndex.toFixed(1)}%)，建议向后舱多装载`;
    } else {
      advice = `重心偏后(${cgIndex.toFixed(1)}%)，建议向前舱多装载`;
    }
  }

  return { cgPosition, cgIndex, totalWeight, totalMoment,
           isInLimits, isWarning, limitMin: limits[0],
           limitMax: limits[1], advice };
}
```

```typescript
  const errors: ValidationError[] = [];
  const warnings: ValidationWarning[] = [];

  const uldSpec = ULD_SPECS[uldType];
  const dgrCargo = cargoList.filter(c => c.dgrClass);
  const allDgrCargo = allCargoInFlight.filter(c => c.dgrClass);

  // 1. DGR隔离校验
  for (const cargo of dgrCargo) {
    const separatedClasses = DGR_SEPARATION[cargo.dgrClass]?.cannot_co_load_with || [];
    for (const cls of separatedClasses) {
      const conflict = cargoList.find(c => c.dgrClass === cls);
      if (conflict) {
        errors.push({
          type: 'DGR',
          code: cargo.dgrClass,
          message: `危险品类Class ${cargo.dgrClass}(${cargo.dgrName})与Class ${cls}不得同板装载`
        });
      }
    }
  }

  // 2. 尺寸校验
  for (const cargo of cargoList) {
    if (cargo.length > uldSpec.innerLength || cargo.width > uldSpec.innerWidth) {
      errors.push({
        type: 'DIMENSION',
        cargo: cargo.awb,
        message: `货物${cargo.awb}尺寸(${cargo.length}×${cargo.width}cm)超出ULD ${uldType}内径(${uldSpec.innerLength}×${uldSpec.innerWidth}cm)`
      });
    }
  }

  // 3. 重量校验
  const totalWeight = cargoList.reduce((sum, c) => sum + c.weight, 0);
  if (totalWeight > uldSpec.maxWeight) {
    errors.push({
      type: 'WEIGHT',
      message: `板总重量${totalWeight}kg超过ULD ${uldType}限重${uldSpec.maxWeight}kg`
    });
  }

  // 4. 高度校验
  const stackHeight = cargoList.reduce((sum, c) => sum + c.height, 0);
  if (stackHeight > uldSpec.innerHeight) {
    errors.push({
      type: 'HEIGHT',
      message: `堆叠总高度${stackHeight}cm超过ULD ${uldType}限高${uldSpec.innerHeight}cm`
    });
  }

  // 5. 体积利用率警告
  const totalVolume = cargoList.reduce((sum, c) => sum + c.volume, 0);
  const volumeUtil = totalVolume / uldSpec.maxVolume;
  if (volumeUtil < 0.5) {
    warnings.push({
      type: 'LOW_UTIL',
      message: `体积利用率仅${(volumeUtil * 100).toFixed(1)}%，建议增加货物`
    });
  }

  return { valid: errors.length === 0, errors, warnings };
}
```
